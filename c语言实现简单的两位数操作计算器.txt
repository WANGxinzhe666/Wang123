#define _CRT_SECURE_NO_WARNINGS 1
#include<stdio.h>
#include<math.h>
#include<string.h>
int add(int x, int y) {
	return x + y;
}
int sub(int x,int y) {
	return x - y;

}
int mul(int x, int y) {
	return x * y;
}
int div(int x, int y) {
	return x / y;
}
int zuoyi(int x, int y) {
	return x << y;
}
int youyi(int x, int y) {
	return x >> y;
}
void menu() {
	printf("******************************************\n");
	printf("*****    1:add    2:sub                ***\n");
	printf("*****    3:mul    4:div                ***\n");
	printf("*****    5:zuoyi  6:youyi              ***\n");
	printf("*****    0.exit                        ***\n");
	printf("******************************************\n");

}
int main() {
	int x = 0;//操作的两个数字
	int y = 0;

	int input = 0;//选择数字
	int ret = 0; //两位数字的结果
	int(*ptr[])(int x, int y) = { 0, add, sub, mul, div,zuoyi,youyi }; //用函数指针存放函数的地址
	do {
		menu();
		printf("请输入 :");
		scanf("%d", &input);
		if (input == 0) {
			printf("退出计算器 ！！");
		}
		else {
			if (input >= 1 && input <= 6) {
				printf("请输入2个操作数：>");
				scanf("%d %d", &x, &y);
				ret =(*ptr[input])(x, y);
				printf("%d\n", ret);
			}
			else {
				printf("输入错误 请重新输入！！");
			}
		}




	} while (input);

}
